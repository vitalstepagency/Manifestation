typescriptclass StreakIntelligence {
calculateStreak(habit: Habit, timezone: string): StreakData {
const now = DateTime.now().setZone(timezone);
const lastCompletion = DateTime.fromISO(habit.lastCompletedAt).setZone(timezone);

    // Check if completed today
    const completedToday = lastCompletion.hasSame(now, 'day');

    // Check if streak broken
    const daysSinceCompletion = now.diff(lastCompletion, 'days').days;
    const streakBroken = daysSinceCompletion > 1;

    if (streakBroken) {
      return {
        current: completedToday ? 1 : 0,
        best: Math.max(habit.bestStreak, habit.currentStreak),
        status: 'broken',
        message: this.getBrokenMessage(habit.currentStreak)
      };
    }

    return {
      current: completedToday
        ? habit.currentStreak
        : habit.currentStreak,
      best: habit.bestStreak,
      status: this.getStreakStatus(habit.currentStreak),
      message: this.getMotivationalMessage(habit.currentStreak),
      milestone: this.getNextMilestone(habit.currentStreak)
    };

}

    const hoursSinceCompletion = DateTime.now()
      .diff(DateTime.fromISO(habit.lastCompletedAt), 'hours').hours;

    if (userBehavior.currentEnergy < 2) {

    // Historical pattern risk
    const typicalCompletionHour = this.getTypicalCompletionTime(habit);
    const currentHour = DateTime.now().hour;

    if (currentHour > typicalCompletionHour + 2) {
      riskFactors.push('past_usual_time');
      riskScore += 0.25;
    }

    // Day of week risk
    const weakestDay = this.getWeakestDay(habit);
    if (DateTime.now().weekdayLong === weakestDay) {
      riskFactors.push('historically_difficult_day');
      riskScore += 0.15;
    }

    return {
      riskScore,
      riskFactors,
      preventionStrategy: this.getPreventionStrategy(riskFactors),
      notification: riskScore > 0.5
        ? this.generateUrgentNotification(habit, riskFactors)
        : null
    };
